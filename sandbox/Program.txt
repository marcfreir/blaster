                var lexer = new Lexer(inputLine);
                while (true)
                {
                    var token = lexer.NextToken();
                    if (token.Kind == SyntaxKind.EndOfFileToken)
                    {
                        break;
                    }
                    Console.Write($"{token.Kind}: '{token.Text}'");

                    if (token.Value != null)
                    {
                        Console.Write($" {token.Value}");
                    }

                    Console.WriteLine();
                }
